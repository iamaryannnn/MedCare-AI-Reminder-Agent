+    function VoiceWaveform({ isListening }) {
+      const canvasRef = React.useRef(null);
+
+      React.useEffect(() => {
+        const canvas = canvasRef.current;
+        if (!canvas) return;
+        const ctx = canvas.getContext("2d");
+        let animationId;
+        let phase = 0;
+
+        const renderWave = () => {
+          ctx.clearRect(0, 0, canvas.width, canvas.height);
+          
+          // Draw multiple sine waves overlapping with different phases, amplitudes, and opacity (Siri-style)
+          const waves = [
+            { amplitude: isListening ? 25 : 4, speed: 1.5, color: "rgba(20, 184, 166, 0.6)" },  // Cyan/Teal
+            { amplitude: isListening ? 15 : 2, speed: -1.0, color: "rgba(16, 185, 129, 0.4)" },  // Green
+            { amplitude: isListening ? 8 : 1, speed: 0.8, color: "rgba(99, 102, 241, 0.3)" }    // Indigo
+          ];
+
+          phase += 0.05;
+
+          waves.forEach(w => {
+            ctx.beginPath();
+            ctx.lineWidth = 2;
+            ctx.strokeStyle = w.color;
+
+            for (let x = 0; x < canvas.width; x++) {
+              // Sine wave centered vertically, pinched at both ends via sine envelope
+              const envelope = Math.sin(x * Math.PI / canvas.width);
+              const y = canvas.height / 2 + Math.sin(x * 0.025 + phase * w.speed) * w.amplitude * envelope;
+              if (x === 0) {
+                ctx.moveTo(x, y);
+              } else {
+                ctx.lineTo(x, y);
+              }
+            }
+            ctx.stroke();
+          });
+
+          animationId = requestAnimationFrame(renderWave);
+        };
+
+        renderWave();
+
+        return () => {
+          cancelAnimationFrame(animationId);
+        };
+      }, [isListening]);
+
+      return (
